Sum of all Transaction operations


input : a single row {Series}

output : a single row {Series}

    if row.name[1] == "Portfolio":
        row['netSumXaction'] = row.drop(["Transaction's Cash In / Out"]).sum()
        row['grSumXaction'] = row.drop(["Transaction's Cash In / Out", "Transaction Charges", "Man Fees", "Administrator Fee"]).sum()

def calcPortReturns(retBalSheet):

    

  

  

  

    

      

        #
      

      
input : Balance Sheet {DataFrame}

cols = [Date, Group, Asset, Operation, Amount]

output : closing Balance Sheet {DataFrame}

    balSheet = retBalSheet.apply(lambda row: sumXactions(row), axis=1)  # balSheet = pivotRetBalSheet.reset_index().set_index('DATE')

    

  

  

  

    

      

        #
      

      
Initialize closing balance df

    

    

      
    closingBal = pd.DataFrame(index=balSheet.stack().unstack().index)
    closingBal = closingBal.assign(**{"grOpen": np.zeros(balSheet.index.size), "netOpen": np.zeros(balSheet.index.size), "grClose": np.zeros(balSheet.index.size), "netClose": np.zeros(balSheet.index.size), "grReturns": np.zeros(balSheet.index.size), "netReturns": np.zeros(balSheet.index.size), "logGrReturns": np.zeros(balSheet.index.size), "logNetReturns": np.zeros(balSheet.index.size)})
    closingBal = closingBal.unstack('GROUP')
    closingBal['grOpen'].iloc[0] = 100000
    closingBal['netOpen'].iloc[0] = 100000

    balSheet = balSheet.unstack()
    closingBal['netClose'].values[:] = closingBal['netOpen'].values[0] + balSheet['netSumXaction'].cumsum().values
    closingBal['grClose'].values[:] = closingBal['grOpen'].values[0] + balSheet['grSumXaction'].cumsum().values
    closingBal['netOpen'].values[1:] = closingBal['netClose'].values[:-1]
    closingBal['grOpen'].values[1:] = closingBal['grClose'].values[:-1]

    closingBal['netReturns'] = closingBal['netClose'].pct_change()
    closingBal['grReturns'] = closingBal['grClose'].pct_change()

    closingBal['logGrReturns'] = np.log(closingBal['grClose'].pct_change() + 1)
    closingBal['logNetReturns'] = np.log(closingBal['netClose'].pct_change() + 1)

    return closingBal
